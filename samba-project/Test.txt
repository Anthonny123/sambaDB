-- CREATE TABLES

CREATE TABLE MAVhabilidad (
    codigo SERIAL PRIMARY KEY NOT NULL,
    nombre_habilidad VARCHAR(30) NOT NULL,
    descripcion VARCHAR(100) NOT NULL
);

CREATE TABLE MAVintegrante_escuela (
    codigo SERIAL PRIMARY KEY NOT NULL,
    primer_nombre VARCHAR(30) NOT NULL,
    segundo_nombre VARCHAR(30),
    primer_apellido VARCHAR(30) NOT NULL,
    segundo_apellido VARCHAR(30) NOT NULL,
    documento_identidad INT,
    apodo VARCHAR(30),
    fecha_nacimiento DATE NOT NULL,
    nacionalidad VARCHAR(30)
);

CREATE TABLE MAVcolor (
    codigo SERIAL PRIMARY KEY NOT NULL,
    nombre_color VARCHAR(30) NOT NULL
);

CREATE TABLE MAVsamba (
    codigo SERIAL PRIMARY KEY NOT NULL,
    letra VARCHAR(1200) NOT NULL,
    nombre_cancion VARCHAR(50) NOT NULL,
    year_carnaval VARCHAR(30) NOT NULL,
    tipo VARCHAR(30) CHECK (tipo IN ('Samba Enredo', 'Samba Bloco', 'Manchinha', 'Frevo', 'Maracatu', 'Samba Reggae'))
);

CREATE TABLE MAVrol (
    codigo SERIAL PRIMARY KEY NOT NULL,
    nombre_rol VARCHAR(30) NOT NULL,
    descripcion VARCHAR(100) NOT NULL
);

CREATE TABLE MAVpatrocinador_natural (
    codigo SERIAL PRIMARY KEY NOT NULL,
    documento_identidad INT NOT NULL,
    primer_nombre VARCHAR(30) NOT NULL,
    segundo_nombre VARCHAR(30),
    primer_apellido VARCHAR(30) NOT NULL,
    segundo_apellido VARCHAR(30),
    fecha_nacimiento DATE NOT NULL,
    contacto_email VARCHAR(30) NOT NULL
);

CREATE TABLE MAVpatrocinador_juridico (
    codigo SERIAL PRIMARY KEY NOT NULL,
    nombre_empresa VARCHAR(30) NOT NULL,
    contacto_email VARCHAR(40) NOT NULL,
    mision VARCHAR(250)
);

CREATE TABLE MAVlugar ( 
    codigo SERIAL PRIMARY KEY NOT NULL,
    nombre VARCHAR(30) NOT NULL,
    tipo VARCHAR(30) CHECK(tipo IN ('Ciudad', 'Estado')),
    region VARCHAR(30) CHECK(region IN ('Norte', 'Noroeste', 'Oeste', 'Suroeste', 'Sur', 'Sureste', 'Este', 'Nordeste')),

	cod_lugar_padre INT,
	CONSTRAINT fk_lugar_padre FOREIGN KEY (cod_lugar_padre) REFERENCES MAVlugar(codigo)
);

CREATE TABLE MAVpremio_especial (
    codigo SERIAL PRIMARY KEY NOT NULL,
    nombre VARCHAR(30) NOT NULL,
    tipo VARCHAR(30) CHECK(tipo IN('Escuela', 'Individual')),
    descripcion VARCHAR(100) NOT NULL,
	cod_lugar INT NOT NULL,
	CONSTRAINT fk_cod_lugar FOREIGN KEY (cod_lugar) REFERENCES MAVlugar(codigo)
);


CREATE TABLE MAVescuela_samba (    
    codigo SERIAL PRIMARY KEY NOT NULL,
    historia VARCHAR(600) NOT NULL,
    sede VARCHAR(30) NOT NULL,
    fecha_fundacion DATE NOT NULL,
    nombre_escuela VARCHAR(30) NOT NULL,
	cod_lugar INT NOT NULL,
	CONSTRAINT fk_cod_lugar FOREIGN KEY (cod_lugar) REFERENCES MAVlugar(codigo)
);


CREATE TABLE MAVtelefono ( 
    codigo_internacional INT NOT NULL,
    codigo_area INT NOT NULL,
    numero_telefono INT NOT NULL,
	cod_escuela_samba INT, 
	cod_patrocinador_j INT,
	cod_patrocinador_n INT,
	
        
	CONSTRAINT pk_telefono PRIMARY KEY (codigo_internacional, codigo_area, numero_telefono),

	CONSTRAINT fk_telefono_escuela foreign key (cod_escuela_samba) references MAVescuela_samba(codigo),
	CONSTRAINT fk_telefono_patrocinador_juridico foreign key (cod_patrocinador_j) references MAVpatrocinador_juridico(codigo),
	CONSTRAINT fk_telefono_patrocinador_natural foreign key (cod_patrocinador_n) references MAVpatrocinador_natural(codigo)
);

CREATE TABLE MAVhist_integrante (
    fecha_inicio DATE NOT NULL,
    fecha_fin DATE,
    autoridad CHAR(2) CHECK(autoridad IN ('Si', 'No')),

	cod_escuela_samba INT NOT NULL,
	cod_integrante_escuela INT NOT NULL,
	
	
	CONSTRAINT pk_hist_integrante PRIMARY KEY (cod_escuela_samba,cod_integrante_escuela, fecha_inicio),
	CONSTRAINT fk_escuela FOREIGN KEY (cod_escuela_samba) REFERENCES MAVescuela_samba(codigo),
	CONSTRAINT fk_integrante FOREIGN KEY (cod_integrante_escuela) REFERENCES MAVintegrante_escuela(codigo)
);

CREATE TABLE MAVevento_sem_anual (
    codigo SERIAL NOT NULL,
    nombre VARCHAR(30) NOT NULL,
    descripcion VARCHAR(60),
    tipo VARCHAR(30) CHECK (tipo IN ('Noche samba', 'Otro')),
    fecha_inicio DATE NOT NULL,
    fecha_fin DATE NOT NULL,
    asistencia INT,
    costo_unitario_reales INT NOT NULL,

	cod_escuela_samba INT NOT NULL,
	CONSTRAINT pk_evento_sem_anual PRIMARY KEY (cod_escuela_samba,codigo),
	CONSTRAINT fk_escuela_samba FOREIGN KEY (cod_escuela_samba) REFERENCES MAVescuela_samba(codigo)
);

CREATE TABLE MAVhist_patrocinador ( 
    codigo SERIAL NOT NULL,
    fecha_inicio DATE NOT NULL,
    fecha_fin DATE,

	cod_escuela_samba INT NOT NULL,
	cod_patrocinador_j INT,
	cod_patrocinador_n INT,
	
	CONSTRAINT pk_hist_patrocinador PRIMARY KEY (cod_escuela_samba,codigo),
	CONSTRAINT fk_cod_escuela_samba FOREIGN KEY (cod_escuela_samba) REFERENCES MAVescuela_samba(codigo),
	CONSTRAINT fk_cod_patrocinador_j FOREIGN KEY (cod_patrocinador_j) REFERENCES MAVpatrocinador_juridico(codigo),
	CONSTRAINT fk_cod_patrocinador_n FOREIGN KEY (cod_patrocinador_n) REFERENCES MAVpatrocinador_natural(codigo)
);

CREATE TABLE MAVpatrocinador_donacion ( 
    codigo SERIAL NOT NULL,
    monto_reales INT NOT NULL,
    fecha_donacion DATE NOT NULL,

	cod_escuela_samba INT NOT NULL,
	cod_hist_patrocinador INT NOT NULL,

	CONSTRAINT pk_patrocinador_donacion PRIMARY KEY (cod_escuela_samba,cod_hist_patrocinador,codigo),
	CONSTRAINT fk_cod_hist_patrocinador FOREIGN KEY (cod_hist_patrocinador,cod_escuela_samba) REFERENCES MAVhist_patrocinador(cod_escuela_samba,codigo)
	
);


CREATE TABLE MAVhist_titulo_escuela (
    ano INT NOT NULL,
    grupo VARCHAR(30),
    monto_ganado_reales INT,

	cod_escuela_samba INT NOT NULL,
	CONSTRAINT pk_hist_titulo_escuela PRIMARY KEY (cod_escuela_samba,ano),
	CONSTRAINT fk_cod_escuela_samba FOREIGN KEY (cod_escuela_samba) REFERENCES MAVescuela_samba(codigo)
);


CREATE TABLE MAVorganizacion_carnaval (
    ano INT NOT NULL,
    cod_escuela_samba INT NOT NULL,
	cod_integrante_escuela INT NOT NULL,
	fecha_inicio DATE NOT NULL,
    cod_rol INT NOT NULL,

	CONSTRAINT pk_organizacion_carnaval PRIMARY KEY (cod_escuela_samba,cod_integrante_escuela,fecha_inicio,cod_rol,ano),
	CONSTRAINT fk_cod_hist_integrante FOREIGN KEY (cod_escuela_samba,cod_integrante_escuela,fecha_inicio) REFERENCES MAVhist_integrante(cod_escuela_samba,cod_integrante_escuela,fecha_inicio),
	CONSTRAINT fk_cod_rol FOREIGN KEY (cod_rol) REFERENCES MAVrol(codigo)
);

CREATE TABLE MAVganador (
    ano INT NOT NULL,
	cod_premio_especial INT NOT NULL,
	
	fecha_inicio DATE,
	cod_escuela_samba_int INT,
	cod_integrante_escuela INT,
	
	cod_escuela_samba INT,

	CONSTRAINT pk_ganador PRIMARY KEY (cod_premio_especial,ano),
	CONSTRAINT fk_cod_premio_especial FOREIGN KEY (cod_premio_especial) REFERENCES MAVpremio_especial(codigo),
	CONSTRAINT fk_hist_integrante FOREIGN KEY (cod_escuela_samba_int,cod_integrante_escuela,fecha_inicio) REFERENCES MAVhist_integrante(cod_escuela_samba,cod_integrante_escuela,fecha_inicio),
	CONSTRAINT fk_cod_escuela_samba FOREIGN KEY (cod_escuela_samba) REFERENCES MAVescuela_samba(codigo)
);

CREATE TABLE MAVC_E (
	cod_escuela_samba INT NOT NULL,
	cod_color INT NOT NULL,

	CONSTRAINT pk_c_e PRIMARY KEY (cod_escuela_samba,cod_color),
	CONSTRAINT fk_cod_escuela_samba FOREIGN KEY (cod_escuela_samba) REFERENCES MAVescuela_samba(codigo),
	CONSTRAINT fk_cod_color FOREIGN KEY (cod_color) REFERENCES MAVcolor(codigo));

CREATE TABLE MAVautor (
	cod_samba INT NOT NULL,
	
	cod_integrante_escuela INT NOT NULL,
	cod_escuela_samba INT NOT NULL,
	fecha_inicio DATE NOT NULL,

	CONSTRAINT pk_autor PRIMARY KEY (cod_samba,cod_integrante_escuela,cod_escuela_samba,fecha_inicio),
	CONSTRAINT fk_hist_integrante FOREIGN KEY (cod_escuela_samba,cod_integrante_escuela,fecha_inicio) REFERENCES MAVhist_integrante(cod_escuela_samba,cod_integrante_escuela,fecha_inicio),
	CONSTRAINT fk_cod_samba FOREIGN KEY (cod_samba) REFERENCES MAVsamba(codigo));

CREATE TABLE MAVI_H (
	cod_integrante_escuela INT NOT NULL,
	cod_habilidad INT NOT NULL,
	
	CONSTRAINT pk_I_H PRIMARY KEY (cod_integrante_escuela,cod_habilidad),
	CONSTRAINT fk_cod_integrante_escuela FOREIGN KEY (cod_integrante_escuela) REFERENCES MAVintegrante_escuela(codigo),
	CONSTRAINT fk_cod_habilidad FOREIGN KEY (cod_habilidad) REFERENCES MAVhabilidad(codigo));


CREATE TABLE MAVparentesco ( 
    nombre_parentesco VARCHAR(30) CHECK(nombre_parentesco IN ('Mama', 'Papa', 'Hermano', 'Hermana', 'Tio', 'Tia', 'Sobrino', 'Sobrina', 'Abuela', 'Abuelo', 'Nieto', 'Nieta', 'Otro')),
	cod_integrante_escuela1 INT NOT NULL,
	cod_integrante_escuela2 INT NOT NULL,
	
	CONSTRAINT pk_parentesco PRIMARY KEY (cod_integrante_escuela1,cod_integrante_escuela2),
	CONSTRAINT fk_cod_integrante_escuela1 FOREIGN KEY (cod_integrante_escuela1) REFERENCES MAVintegrante_escuela(codigo),
	CONSTRAINT fk_cod_integrante_escuela2 FOREIGN KEY (cod_integrante_escuela2) REFERENCES MAVintegrante_escuela(codigo)

SELECT COUNT(*) AS total_sambas FROM MAVSamba; --Obtener la cantidad total de sambas:
SELECT * FROM MAVSamba ORDER BY year_carnaval DESC; --Obtener las sambas ordenadas por año de carnaval de forma descendente:
SELECT * FROM MAVSamba WHERE nombre_cancion LIKE '%samba%'; --Obtener las sambas cuyo nombre de canción contiene la palabra "samba":


-- CONSULTAS TABLA INTEGRANTE_ESCUELA
SELECT * FROM MAVIntegrante_Escuela;
SELECT * FROM MAVIntegrante_Escuela WHERE nacionalidad = 'Venezuela'; --Integrantes por nacionalidad
SELECT * FROM MAVIntegrante_Escuela WHERE primer_apellido LIKE '%Silva%'; --Integrantes que contiene la palabra Silva
SELECT primer_nombre, primer_apellido, fecha_nacimiento, 
       AGE(CURRENT_DATE, fecha_nacimiento) AS edad 
FROM MAVIntegrante_Escuela; --Obtener la edad de los integrantes de la escuela:
SELECT * FROM MAVIntegrante_Escuela WHERE segundo_nombre IS NULL; --Obtener los integrantes cuyo segundo nombre es nulo:
SELECT * FROM MAVIntegrante_Escuela 
WHERE fecha_nacimiento < '1990-01-01' AND nacionalidad = 'Argentina'; --Obtener los integrantes que nacieron antes de 1990 y tienen nacionalidad "Argentina":

SELECT * FROM MAVPatrocinador_Juridico WHERE contacto_email IS NOT NULL; --Obtener los patrocinadores jurídicos que tienen un correo electrónico de contacto:

SELECT AVG(EXTRACT(YEAR FROM AGE(CURRENT_DATE, fecha_nacimiento))) AS promedio_edad
